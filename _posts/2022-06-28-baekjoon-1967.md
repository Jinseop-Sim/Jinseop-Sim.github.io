## 백준 1967 : 트리의 지름

내용 추가예정.

{% highlight cpp %}
#include <iostream>
#include <vector>
using namespace std;
#define MAX 10001

vector<pair<int, int>> tree[MAX];
bool visit[MAX] = { 0, };
int n = 0, max_dist = 0, max_idx;

void clear_vis() {
    max_dist = 0;
    for (int i = 0; i < MAX; i++)
        visit[i] = false;
}

void dfs(int start, int dist) {
    if (visit[start]) return;

    visit[start] = true;
    if (max_dist < dist) {
        max_dist = dist;
        max_idx = start;
    }

    for (int i = 0; i < tree[start].size(); i++)
        dfs(tree[start][i].first, dist + tree[start][i].second);
}

int main() {
    cin >> n;
    for (int i = 0; i < n-1; i++) {
        int src, dest, weight;
        cin >> src >> dest >> weight;
        tree[src].push_back(make_pair(dest, weight));
        tree[dest].push_back(make_pair(src, weight));
    }
    
    dfs(1, 0);

    clear_vis();

    dfs(max_idx, 0);
    cout << max_dist;
    return 0;
}
